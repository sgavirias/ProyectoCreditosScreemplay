        let urlPoliticas = "/autogestionnegozia/politicas/";

        function mostrarPolitica(archivo) {
            let url = urlPoliticas + archivo;
            var visor = document.getElementById("visorPolitica");
            var clone = visor.cloneNode(true);
            clone.setAttribute('src', url);
            visor.parentNode.replaceChild(clone, visor);
            $("#modalPoliticas").modal("show");

        function limpiarDatos() {
            $("#jornada").val("");
            $("#resultadoMaterias").empty();
            $("#btnGuardarLiquidacion").prop("disabled", true);
            $("#divBotones").hide();
            $("#divPoliticaPrograma").hide();
            aplicaPolitica = false;
        }

        function validarEstudiante() {

            var continuar = true;

            $(".requerido").each(function () {
                if (continuar) {
                    continuar = $(this).requerido();
                } else {
                    $(this).requerido();
                }
            });

            if (!continuar) return false;

            let id = $("#identificacion").val();

            $.ajax({
                url: '/autogestionnegozia/Home/ValidarEstudiante',
                data: { identificacion: id },
                method: 'GET',
                dataType: 'json',
                success: function (result) {

                    if (result.valido === true) {

                        let info = result.info;
                        if (info != null) {
                            asignarDatosBasicos(info);
                        }

                        $("#datosBasicos").show("slow");
                        $("#identificacion, #tipoIdentificacion").prop("disabled", true);
                        $("#pnlDatos").show("slow");
                    }
                    else {
                        Swal.fire({
                            type: 'info',
                            title: 'Información',
                            text: 'Este formulario es solo para estudiantes Nuevos, para realizar tu matricula presiona el boton Matrícula para Antiguos',
                            showConfirmButton: false,
                            footer: '&lt;a href="http://diamante.cesde.edu.co/Matriculas/" class="btn btn-info btn-block"&gt;Matricula para Antiguos&lt;/a&gt;'
                        })
                    }
                }
            });
        }

        function asignarDatosBasicos(datos) {
            $("#primerNombre").val(datos.nombre1Estudiante);
            $("#segundoNombre").val(datos.nombre2Estudiante);
            $("#primerApellido").val(datos.apellido1Estudiante);
            $("#segundoApellido").val(datos.apellido2Estudiante);
            $("#telefonoFijo").val(datos.telefono1Estudiante);
            $("#telefonoCelular").val(datos.telefonoCelular);
            $("#correo").val(datos.emailEstudiante);
        }

        function guardarDatos() {

            var aceptaGeneral = $("#aceptaGeneral").prop("checked");
            var aceptaTratamiento = $("#aceptaTratamiento").prop("checked");
            var aceptaEspecifica = $("#aceptaEspecifica").prop("checked");
            if (aceptaGeneral === false || aceptaTratamiento === false || (aplicaPolitica === true &amp;&amp; aceptaEspecifica === false)) {
                Swal.fire({
                    type: 'info',
                    title: 'Importante!',
                    text: 'Debes leer y aceptar las políticas para poder continuar con el proceso de matrícula',
                })

                return;
            }

            var continuar = true;

            $(".requerido1").each(function () {
                if (continuar) {
                    continuar = $(this).requerido();
                } else {
                    $(this).requerido();
                }
            });

            if (!continuar) return false;

            let datos = {
                identificacion: $("#identificacion").val(),
                tipoIdentificacion: $("#tipoIdentificacion").val(),
                nombre1: $("#primerNombre").val(),
                nombre2: $("#segundoNombre").val(),
                apellido1: $("#primerApellido").val(),
                apellido2: $("#segundoApellido").val(),
                telefonoFijo: $("#telefonoFijo").val(),
                telefonoCelular: $("#telefonoCelular").val(),
                correo: $("#correo").val(),
                sede: $("#sede").val(),
                jornada: $("#jornada").val(),
                programa: $("#programa").val(),
            };
            
            $.ajax({
                url: '/autogestionnegozia/Home/GuardarDatos',
                data: datos,
                method: 'POST',
                dataType: 'json',
                success: function (result) {

                    if (result === true) {
                        Swal.fire({
                            type: 'success',
                            title: 'Muy bien!',
                            text: 'Tus datos se han guardado correctamente, ahora puedes elegir el método de pago de tu matrícula',
                        })

                        $("#divBotones").show("slow");
                        $("#btnGuardarLiquidacion").prop("disabled", true);
                    }
                    else {
                        Swal.fire({
                            type: 'error',
                            title: 'Lo sentimos...',
                            text: 'Ha ocurrido un error guardando la asesoria',
                        })
                    }
                }
            });

        }

        function obtenerProgramas() {

            //Se limpia el combo
            $("#programa").empty();
            limpiarDatos();

            const sede = $("#sede").val();
            if (sede === "") {
                return;
            }

            $.ajax({
                url: '/autogestionnegozia/Home/ObtenerProgramas',
                data: { idSede: sede },
                method: 'GET',
                dataType: 'json',
                success: function (data) {

                    $('#programa').append($('&lt;option&gt;', {
                        value: '',
                        text: 'Seleccione un Programa'
                    }));

                    $.each(data, function (i, item) {
                        $('#programa').append($('&lt;option&gt;', {
                            value: item.codigoPrograma,
                            text: item.nombreAutogestion
                        }));
                    });
                }
            });
        }

        function obtenerSubmodulosTarifa() {

            const programa = $("#programa").val();
            if (programa === "") {
                return;
            }

            const jornada = $("#jornada").val();
            if (jornada === "") {
                return;
            }

            $.ajax({
                url: '/autogestionnegozia/Home/ObtenerSubmodulosTarifa',
                data: { codigoPrograma: programa, codigoJornada: jornada },
                method: 'GET',
                dataType: 'json',
                success: function (result) {

                    $("#resultadoMaterias").empty();
                    $("#btnGuardarLiquidacion").prop("disabled", true);

                    if (result.esValido === true) {
                        $("#resultadoMaterias").html(result.html);
                        $("#btnGuardarLiquidacion").prop("disabled", false);
                    } else {
                        Swal.fire({
                            type: 'error',
                            title: 'Lo sentimos...',
                            text: 'No se han encontrado submódulos con cupo disponible en la jornada seleccionada, por favor seleccione otra jornada e intente nuevamente',
                        })
                    }
                }
            });
        }

        function procesarPago() {

            $.ajax({
                url: '/autogestionnegozia/Home/GenerarDatosPago',
                method: 'GET',
                dataType: 'html',
                success: function (result) {

                    $("#pnlDatosPago").html("");
                    $("#pnlDatosPago").html(result);
                    $("#frmPago").submit();
                }
            });

        }

        function validarPoliticaPrograma() {

            limpiarDatos();

            const programa = $("#programa").val();
            if (programa === "") {
                return;

            $.ajax({
                url: '/autogestionnegozia/Home/AplicaPoliticaPrograma',
                data: { codigoPrograma: programa },
                method: 'GET',
                dataType: 'json',
                success: function (result) {

                    if (result === true) {
                        $("#divPoliticaPrograma").show();
                        aplicaPolitica = true;
                    }
                }
            });
        }
